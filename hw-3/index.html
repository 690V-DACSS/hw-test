library(dplyr)

locale_counts <- data %>%
  count(Locale) %>%
  arrange(desc(n))

Categorical Variable: Locale

This plot shows the number of schools by locale in Massachusetts.

ggplot(locale_counts, aes(x = reorder(Locale, -n), y = n)) +
  geom_col(fill = "#2C6BAA") +
  labs(
    title = "Number of Schools by Locale in Massachusetts",
    x = "School Locale",
    y = "Number of Schools",
    caption = "Source: NCES"
  ) +
  theme_minimal() +
  theme(axis.text.x = element_text(angle = 45, hjust = 1))
